% \HFIGS{Label}{Title}{Title1}{Title2}{PNG1}{PNG2}
\newcommand{\HFIGS}[6]{
\begin{figure}[htbp]
  \centering
  \subfloat[#3]{
    {\scalebox{\hscale}{\includegraphics*[bb=0pt 0pt 800pt 800pt]
    {./figs/#5.png}}}} 
  \subfloat[#4]{
    {\scalebox{\hscale}{\includegraphics*[bb=0pt 0pt 800pt 800pt]
    {./figs/#6.png}}}}
  \caption{#2}
  \label{#1}
\end{figure}
}

% \VFIGS{Label}{Title}{Title1}{Title2}{PNG1}{PNG2}
\newcommand{\VFIGS}[6]{

  \caption{#2: {\em #3} and {\em #4}}
  \label{#1}
  \end{center}
\end{figure}
